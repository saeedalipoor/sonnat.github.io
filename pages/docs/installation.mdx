import SonnatInitializer from "@sonnat/ui/styles/SonnatInitializer";\n
const App = () => (
  <SonnatInitializer>
    <CssBaseline />
    <MyComponent />
  </SonnatInitializer>
);`

    <div data-notebox style={{ marginBottom: "1rem" }}>
      <strong>&lt;CssBaseline&gt;</strong> provides basic CSS support, which is

      <br />
      1- Exposes CSS variables to the <strong>:root</strong> so it can be used later
      in CSS.
      <br />
      2- Adds <strong>dir</strong> attribute (based on the provided theme) to the{" "}
      <strong>body</strong> element.
      <br />
      3- Adds <strong>fontFamily</strong> css property (based on the provided theme)
      to the body.
      <br />
      4- Adds <strong>color</strong> and <strong>backgroundColor</strong> css properties
      (based on the provided theme) to the body.
      <br />
      To learn more about the <strong>&lt;CssBaseline&gt;</strong> component, read{" "}
      <a href="/docs/components/CssBaseline" title="CssBaseline Doc">
        this doc
