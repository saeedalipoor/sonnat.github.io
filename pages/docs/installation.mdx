<h3>
  <span>Adding Sonnat to existing project</span>
  <AnchorButton anchorId="adding-sonnat-to-existing-project" />
</h3>

<div data-notebox style={{ marginTop: "0.5rem", marginBottom: "1rem" }}>
  Please note that{" "}
  <a
    href="https://www.npmjs.com/package/react"
    target="_blank"
    rel="noopener noreferrer"
    title="ReactJs"
  >
    react
  </a>{" "}
  >= 17.0.2 and{" "}
  <a
    href="https://www.npmjs.com/package/react-dom"
    target="_blank"
    rel="noopener noreferrer"
    title="ReactJs"
  >
    react-dom
  </a>{" "}
  >= 17.0.2 are peer dependencies.
</div>

<ol>
  <li>
    Run the following script to install and save in your{" "}
    <inlineCode>package.json</inlineCode> dependencies:
    <code
      className="language-bash"
      children={`# with npm
npm install @sonnat/ui\n
# or with yarn
yarn add @sonnat/ui`}
    />
  </li>
  <li>
    Initialize <inlineCode>@sonnat/ui</inlineCode> in your project:
    <code
      className="language-jsx"
      children={
        `` +
        `import CssBaseline from "@sonnat/ui/CssBaseline";
import SonnatInitializer from "@sonnat/ui/styles/SonnatInitializer";\n
const App = () => (
  <SonnatInitializer>
    <CssBaseline />
    <MyComponent />
  </SonnatInitializer>
);`
      }
    />
    <div data-notebox style={{ marginBottom: "1rem", marginTop: "-1rem" }}>
      Check out the documentations of{" "}
      <a
        href="/docs/styling#sonnat-initializer"
        title="SonnatInitializer documentation"
      >
        <strong>&lt;SonnatInitializer&gt;</strong>
      </a>{" "}
      and{" "}
      <a href="/docs/components/CssBaseline" title="CssBaseline documentation">
        <strong>&lt;CssBaseline&gt;</strong>
      </a>
      , to learn more about them.
    </div>
  </li>
  <li>
    Use <inlineCode>@sonnat/ui</inlineCode> components and utilities anywhere in
    your React-Tree:
    <code
      className="language-jsx"
      children={
        `` +
        `import Button from "@sonnat/ui/Button";\n
const MyComponent = () => <Button label="Text" />;`
      }
    />
  </li>
  <li>
    If your project needs to be rendered on the server as well, follow{" "}
    <a href="/docs/server-side-rendering">this guide</a>.
  </li>
</ol>
